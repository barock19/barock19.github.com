    def ck_upload

       uploaded_io = params[:upload]
       cKEditor    = params[:CKEditor] ;
        funcNum     = params[:CKEditorFuncNum] ;
        langCode    = params[:langCode] ;
       ext =  uploaded_io.content_type.split('/').last
       file_name = Digest::SHA1.hexdigest(Time.now.to_s)+'.'+ext
       file_path = Rails.root.join('public', 'uploads', file_name)
       message = ''

       begin
         File.open(file_path, 'wb') do |file|
           file.write(uploaded_io.read)
         end
       rescue Exception => e
         message = e
       end
       url = public_url file_path
       render :text => "<script type='text/javascript'> window.parent.CKEDITOR.tools.callFunction('#{funcNum}', '#{url}', '#{message}')</script>";

    end
